- USERS : {
	'id_user',  // Unique identifier for the user
	'rol': {'student','admin'},  // Role of the user
	'user': integer (UK),  // Unique, typically an identity document number
	'password': string,  // User's password
	'position': string,  // Position or role in the organization
	'cellphone_number': integer,  // Contact number
	'name': string,  // User's first name
	'last_name': string,  // User's last name
	'email': string (UK),  // Unique, email address of the user
	'information': long,  // Additional information about the user
	'city': string,  // City where the user is based
	'photo' -> id_photo,  // Reference to the user's photo
	'projects' -> list[]  (id_projects),  // List of project IDs the user is associated with
	'publications' -> list[] (id_publications)  // List of publication IDs the user is associated with
}
